Building a bot made simple

            
var BotKit = require('./lib/botkit');

var config = {
  client: new BotKit.CLIClient(),
  router: function(bot, request, callback) {

      var action = {
        name: "text",
        payload: {
          text: "Hello World"
        }
      }

      var newContext = bot.dispatch(action, request)
      callback(newContext)
  }
};

var bot = new BotKit.Bot(config);
bot.start();
